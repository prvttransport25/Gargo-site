    <!-- JavaScript for Modal Functionality -->
    <script>
        const modal = document.getElementById('quote-modal');
        const modalContent = document.getElementById('modal-content');
        const quoteSuccess = document.getElementById('quote-success');

        function openModal() {
            quoteSuccess.classList.add('hidden'); // Reset success message
            modal.classList.remove('hidden');
            modal.classList.add('flex');
            // Animate in
            setTimeout(() => {
                modal.classList.add('opacity-100');
                modalContent.classList.add('scale-100', 'opacity-100');
                modalContent.classList.remove('scale-95', 'opacity-0');
            }, 10);
        }

        function closeModal() {
            // Animate out
            modal.classList.remove('opacity-100');
            modalContent.classList.remove('scale-100', 'opacity-100');
            modalContent.classList.add('scale-95', 'opacity-0');

            setTimeout(() => {
                modal.classList.add('hidden');
                modal.classList.remove('flex');
            }, 300);
        }

        // Close modal when clicking outside the content area
        modal.addEventListener('click', (e) => {
            if (e.target === modal) {
                closeModal();
            }
        });

        // Simple form submission handler
        function submitQuoteForm() {
            // In a real application, you would send this data to a backend API.
            // For this example, we just simulate success and log the new data points.
            const fullName = document.getElementById('fullName').value;
            const phoneNumber = document.getElementById('phoneNumber').value;
            const pickup = document.getElementById('pickup').value;
            const delivery = document.getElementById('delivery').value;
            const pickupDate = document.getElementById('pickupDate').value;
            const deliverySpeed = document.getElementById('deliverySpeed').value; // New field

            console.log("--- New Quote Request Submitted ---");
            console.log(`- Full Name: ${fullName}`);
            console.log(`- Phone Number: ${phoneNumber}`);
            console.log(`- Pick Up Location: ${pickup}`);
            console.log(`- Delivery Location: ${delivery}`);
            console.log(`- Pick Up Date: ${pickupDate}`);
            console.log(`- Delivery Type: ${deliverySpeed}`);
            
            // Clear form (or just show success)
            document.getElementById('fullName').value = '';
            document.getElementById('phoneNumber').value = '';
            document.getElementById('pickup').value = '';
            document.getElementById('delivery').value = '';
            document.getElementById('pickupDate').value = '';
            document.getElementById('deliverySpeed').value = '';

            // Hide the form and show the success message
            const form = modalContent.querySelector('form');
            form.classList.add('hidden');
            quoteSuccess.classList.remove('hidden');

            // You might choose to automatically close the modal after a few seconds
            setTimeout(() => {
                form.classList.remove('hidden'); // Reset form for next attempt
                quoteSuccess.classList.add('hidden');
                closeModal();
            }, 4000); 
        }
    </script>
